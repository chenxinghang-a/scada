### 4.1.1 Modbus客户端实现

Modbus客户端模块基于pymodbus库实现，封装了Modbus TCP和Modbus RTU两种通信方式。客户端类提供了统一的读取接口，支持多种数据类型的解码。

核心代码如下：

```python
from pymodbus.client import ModbusTcpClient
import struct

class ModbusClient:
    def __init__(self, config):
        self.host = config.get('host')
        self.port = config.get('port', 502)
        self.slave_id = config.get('slave_id', 1)
        self.client = None
        self.connected = False
    
    def connect(self):
        self.client = ModbusTcpClient(self.host, port=self.port)
        self.connected = self.client.connect()
        return self.connected
    
    def read_holding_registers(self, address, count):
        result = self.client.read_holding_registers(
            address=address, count=count, slave=self.slave_id
        )
        if result.isError():
            return None
        return result.registers
    
    def decode_float32(self, registers):
        raw = struct.pack('>HH', registers[0], registers[1])
        return struct.unpack('>f', raw)[0]
```

### 4.1.2 OPC UA客户端实现

OPC UA客户端基于asyncua库实现，采用异步架构，在独立线程中运行事件循环，通过订阅机制实现数据变更实时推送。

```python
import asyncio
import threading
from asyncua import Client

class OPCUAClient:
    def __init__(self, config):
        self.endpoint = config.get('endpoint')
        self.node_configs = config.get('nodes', [])
        self.username = config.get('username')
        self.password = config.get('password')
        self.connected = False
        self.latest_data = {}
        self._loop = None
        self._thread = None

    def connect(self) -> bool:
        """在独立线程中启动异步事件循环"""
        self._thread = threading.Thread(target=self._run_loop, daemon=True)
        self._thread.start()
        return self.connected

    async def _connect_and_subscribe(self):
        """建立连接并订阅所有配置的节点"""
        self.client = Client(url=self.endpoint)
        if self.username:
            self.client.set_user(self.username)
            self.client.set_password(self.password)

        await self.client.connect()
        self.connected = True
        self.subscription = await self.client.create_subscription(500, self)

        for node_cfg in self.node_configs:
            node = self.client.get_node(node_cfg['node_id'])
            await self.subscription.subscribe_data_change(node)

    def datachange_notification(self, node, val, data):
        """数据变更回调 — asyncua框架自动调用"""
        node_name = self._node_map.get(str(node), str(node))
        self.latest_data[node_name] = {
            'value': val,
            'timestamp': datetime.now()
        }
```

**答辩话术**：OPC UA客户端在独立线程中运行asyncio事件循环，通过asyncua库的订阅机制实现数据变更实时推送。当OPC UA服务器上的节点数据发生变化时，客户端的datachange_notification回调会被自动调用，不需要轮询，大大降低了网络开销。

### 4.1.3 MQTT客户端实现

MQTT客户端基于paho-mqtt库实现，采用发布/订阅模式，支持多主题订阅和JSON数据解析。

```python
import paho.mqtt.client as mqtt
import json

class MQTTClient:
    def __init__(self, config):
        self.broker_host = config.get('host')
        self.broker_port = config.get('port', 1883)
        self.topic_configs = config.get('topics', [])
        self.latest_data = {}

        self.client = mqtt.Client(
            client_id=f"scada_{config.get('id', '')}",
            protocol=mqtt.MQTTv311
        )
        if config.get('username'):
            self.client.username_pw_set(config['username'], config.get('password', ''))
        self.client.on_connect = self._on_connect
        self.client.on_message = self._on_message

    def connect(self) -> bool:
        self.client.connect(self.broker_host, self.broker_port)
        self.client.loop_start()
        return True

    def _on_connect(self, client, userdata, flags, rc):
        """连接成功后自动订阅所有配置的主题"""
        for topic_cfg in self.topic_configs:
            client.subscribe(topic_cfg['topic'], qos=1)

    def _on_message(self, client, userdata, msg):
        """收到消息回调 — 解析JSON并提取数据"""
        topic_cfg = self._find_topic_config(msg.topic)
        if not topic_cfg:
            return
        try:
            data = json.loads(msg.payload.decode('utf-8'))
            value = self._extract_by_path(data, topic_cfg.get('json_path', ''))
            self.latest_data[topic_cfg['name']] = {
                'value': value, 'unit': topic_cfg.get('unit', ''),
                'timestamp': datetime.now()
            }
        except Exception as e:
            logger.error(f"MQTT消息解析失败: {e}")
```

**答辩话术**：MQTT客户端订阅工厂内IoT传感器的主题，实时接收数据。支持多主题订阅，每个主题可以配置独立的JSON路径解析规则。QoS 1级别保证消息至少送达一次，确保数据不丢失。

### 4.1.4 REST HTTP客户端实现

REST HTTP客户端基于requests库实现，对接智能网关和PLC的HTTP API接口。

```python
import requests
import time
import threading

class RESTDeviceClient:
    def __init__(self, config):
        self.base_url = config.get('base_url')
        self.endpoints = config.get('endpoints', [])
        self.poll_interval = config.get('poll_interval', 10)
        self.auth_type = config.get('auth_type', 'none')
        self._session = requests.Session()
        self.latest_data = {}
        self._running = False

    def connect(self) -> bool:
        """配置认证并启动轮询线程"""
        if self.auth_type == 'bearer':
            self._session.headers['Authorization'] = f'Bearer {self.auth_token}'
        elif self.auth_type == 'api_key':
            self._session.headers['X-API-Key'] = self.auth_token
        elif self.auth_type == 'basic':
            self._session.auth = (self.auth_username, self.auth_password)

        self._running = True
        self._thread = threading.Thread(target=self._poll_loop, daemon=True)
        self._thread.start()
        return True

    def read_endpoint(self, endpoint_config):
        """读取单个端点数据"""
        url = self.base_url + endpoint_config['path']
        resp = self._session.request(
            method=endpoint_config.get('method', 'GET'), url=url, timeout=10
        )
        data = resp.json()
        value = self._extract_by_path(data, endpoint_config.get('json_path', ''))
        return value

    def _poll_loop(self):
        """轮询采集循环（独立线程）"""
        while self._running:
            for ep in self.endpoints:
                try:
                    value = self.read_endpoint(ep)
                    self.latest_data[ep['name']] = {
                        'value': value, 'quality': 'good',
                        'timestamp': datetime.now()
                    }
                except Exception as e:
                    self.latest_data[ep['name']]['quality'] = 'bad'
            time.sleep(self.poll_interval)
```

**答辩话术**：REST客户端对接智能网关和PLC的HTTP接口。支持Bearer Token、Basic Auth、API Key三种认证方式。内置JSON路径解析器，可以处理嵌套的API响应。轮询间隔可配置，数据缓存在内存中供上层读取。

### 4.1.5 设备管理器实现（协议工厂模式）

设备管理器负责管理所有设备的配置和连接，采用协议工厂模式，根据设备配置中的protocol字段自动创建对应的协议客户端。

```python
class DeviceManager:
    SUPPORTED_PROTOCOLS = ['modbus_tcp', 'modbus_rtu', 'opcua', 'mqtt', 'rest']

    def _create_client(self, config, simulation_mode):
        """根据protocol字段自动创建对应协议客户端"""
        protocol = config.get('protocol', 'modbus_tcp')

        if protocol in ('modbus_tcp', 'modbus_rtu'):
            if simulation_mode:
                return SimulatedModbusClient(config)
            else:
                return ModbusClient(config)
        elif protocol == 'opcua':
            return OPCUAClient(config)
        elif protocol == 'mqtt':
            return MQTTClient(config)
        elif protocol == 'rest':
            return RESTDeviceClient(config)
        return None

    def add_device(self, config):
        """添加设备并自动创建客户端"""
        device_id = config['id']
        protocol = config.get('protocol', 'modbus_tcp')
        if protocol not in self.SUPPORTED_PROTOCOLS:
            return False
        self.devices[device_id] = config
        client = self._create_client(config, self.simulation_mode)
        if client:
            self.clients[device_id] = client
        return True
```

**答辩话术**：设备管理器是整个采集层的核心，采用协议工厂模式。新增设备只需在配置文件中指定protocol字段，系统自动创建对应的协议客户端。支持运行时动态添加和删除设备，支持热加载配置，不需要重启系统。

### 4.1.6 数据采集器实现（多协议调度）

数据采集器是整个系统的数据枢纽，统一调度所有协议设备的数据采集，区分轮询型（Modbus/REST）和推送型（OPC UA/MQTT）两种模式。

```python
class DataCollector:
    def __init__(self, device_manager, database, alarm_manager):
        self.device_manager = device_manager
        self.database = database
        self.alarm_manager = alarm_manager
        self.data_queue = Queue(maxsize=10000)
        self.running = False

    def start(self):
        """启动数据采集"""
        for device_id, device_config in devices.items():
            protocol = device_config.get('protocol', 'modbus_tcp')
            if protocol in ('opcua', 'mqtt'):
                # OPC UA和MQTT使用事件驱动，只需注册回调
                self._setup_push_device(device_id, device_config)
            else:
                # Modbus和REST使用定时轮询
                self._start_device_collection(device_id, device_config)

    def _setup_push_device(self, device_id, device_config):
        """为推送型设备注册数据回调"""
        client = self.device_manager.get_client(device_id)
        client.add_data_callback(self._on_push_data)

    def _on_push_data(self, device_id, name, value, unit):
        """推送型设备数据回调"""
        self.data_queue.put({
            'device_id': device_id,
            'register_name': name,
            'value': value,
            'unit': unit
        })

    def _process_data(self):
        """数据处理线程：队列 → 存储 → 报警"""
        while self.running:
            data = self.data_queue.get(timeout=1)
            self.database.insert_data(
                data['device_id'], data['register_name'],
                data['value'], datetime.now(), data.get('unit', '')
            )
            self.alarm_manager.check_alarm(
                data['device_id'], data['register_name'],
                data['value'], datetime.now()
            )
```

**答辩话术**：数据采集器是整个系统的数据枢纽。对于Modbus和REST设备，采用定时轮询；对于OPC UA和MQTT设备，采用事件驱动——设备数据变更时主动推送过来。所有数据统一进入线程安全队列，由独立的处理线程负责存储和报警检查，实现了采集和处理的解耦。

```python
class Database:
    def __init__(self, db_path):
        self.db_path = db_path
        Path(db_path).parent.mkdir(parents=True, exist_ok=True)

    def _init_database(self):

                    unit TEXT,
                    timestamp DATETIME NOT NULL

def get_history_data(self, device_id, register_name, 
                     start_time, end_time, interval='1min'):
    if interval == '1min':
        group_format = '%Y-%m-%d %H:%M:00'
    elif interval == '1hour':
        group_format = '%Y-%m-%d %H:00:00'
    
    cursor.execute('''
        SELECT strftime(?, timestamp) as time_bucket,
               AVG(value) as avg_value,
               MIN(value) as min_value,
               MAX(value) as max_value,
               COUNT(*) as sample_count
        FROM history_data
        WHERE device_id = ? AND register_name = ?
            AND timestamp BETWEEN ? AND ?
        GROUP BY time_bucket
        ORDER BY time_bucket
    ''', (group_format, device_id, register_name, start_time, end_time))

    token = request.headers.get('Authorization', '').replace('Bearer ', '')
    user = auth_manager.verify_token(token)
    if not user or user['role'] not in ['admin', 'engineer']:
        return jsonify({'error': '需要管理员或工程师权限'}), 403
    
    data = request.get_json()
    address = data.get('address')
    value = data.get('value')
    
    device_manager = current_app.device_manager
    client = device_manager.get_client(device_id)
    
    # 写入寄存器
    success = client.write_single_register(int(address), int(value))
    
    if success:
        # 记录操作日志
        auth_manager._log_operation(
            user['username'], 'write_register',
            f'设备 {device_id} 写入寄存器 address={address} value={value}'
        )
        return jsonify({'success': True, 'message': f'写入成功'})
    return jsonify({'success': False, 'message': '写入失败'}), 400
```

### 4.3.4 报警引擎实现

报警引擎负责检测数据是否超出阈值，并触发相应的报警。

```python
class AlarmManager:
    def check_alarm(self, device_id, register_name, value, timestamp):
        for rule_id, rule_config in self.rules.items():
            if not rule_config.get('enabled', True):
                continue
            
            if (rule_config['device_id'] == device_id and
                rule_config['register_name'] == register_name):
                
                triggered = self._check_condition(
                    value, rule_config['condition'], rule_config['threshold']
                )
                
                if triggered:
                    self._trigger_alarm(rule_config, device_id, 
                                       register_name, value, timestamp)
    
    def _check_condition(self, value, condition, threshold):
        if condition == 'greater_than':
            return value > threshold
        elif condition == 'less_than':
            return value < threshold
```

### 4.5.1 数据类型转换算法

Modbus寄存器为16位无符号整数，需要根据数据类型进行转换。本系统支持的数据类型转换算法如下：

```python
def decode_float32(self, registers):
    """将两个16位寄存器转换为32位浮点数"""
    raw = struct.pack('>HH', registers[0], registers[1])
    return struct.unpack('>f', raw)[0]

def decode_uint16(self, register):
    """无符号16位整数"""
    return register & 0xFFFF

def decode_int16(self, register):
    """有符号16位整数"""
    if register & 0x8000:
        return register - 0x10000
    return register
```

### 4.5.3 报警阈值判断算法

报警阈值判断支持多种条件类型：

```python
def _check_condition(self, value, condition, threshold):
    if condition == 'greater_than':
        return value > threshold
    elif condition == 'less_than':
        return value < threshold
    elif condition == 'equal_to':
        return abs(value - threshold) < 0.0001
    elif condition == 'greater_equal':
        return value >= threshold
    elif condition == 'less_equal':
        return value <= threshold
```
